    1. Variables & Data Types

    C requires explicit type declarations for all variables. Common types include integers, floating-point numbers, and characters.

    
#include <stdio.h>

int main() {
    // Integer types
    int age = 30;          // Standard integer
    short int small_num = 10; // Smaller integer
    long int large_num = 100000;
    
    // Floating-point types
    float price = 19.99f;  // Single-precision float
    double pi = 3.1415926535; // Double-precision float
    
    // Character type
    char grade = 'A';
    
    // Constants
    const int YEAR = 2024;
    
    printf("Age: %d, Grade: %c, Price: %.2f\n", age, grade, price);
    
    return 0;
}

    2. Control Flow

    Standard control flow structures like `if-else`, `switch`, `for`, `while`, and `do-while` are available.

    
#include <stdio.h>

int main() {
    int i;
    // For loop
    for (i = 0; i < 3; i++) {
        printf("for loop iteration: %d\n", i);
    }
    
    // While loop
    int j = 0;
    while (j < 2) {
        printf("while loop iteration: %d\n", j);
        j++;
    }
    
    // If-else statement
    int score = 85;
    if (score >= 90) {
        printf("Excellent!\n");
    } else if (score >= 70) {
        printf("Good job!\n");
    } else {
        printf("Keep trying!\n");
    }
    
    return 0;
}

    3. Functions

    Functions must be declared before they are used. They consist of a return type, name, parameters, and a body.

    
#include <stdio.h>

// Function declaration (prototype)
int add(int a, int b);

int main() {
    int result = add(10, 20); // Function call
    printf("The sum is: %d\n", result);
    return 0;
}

// Function definition
int add(int a, int b) {
    return a + b;
}

    4. Pointers

    Pointers are variables that store the memory address of another variable. They are a core feature of C.

    
#include <stdio.h>

int main() {
    int var = 20;
    int *p; // Pointer declaration

    p = &var; // Store the address of 'var' in p

    printf("Value of var: %d\n", var);
    printf("Address of var: %p\n", &var);
    printf("Value stored in pointer p: %p\n", p);
    
    // Dereferencing: Accessing the value at the address
    printf("Value at address stored in p: %d\n", *p);
    
    *p = 30; // Change the value of 'var' through the pointer
    printf("New value of var: %d\n", var);
    
    return 0;
}

    5. Arrays & Strings

    Arrays are fixed-size collections of elements of the same type. Strings in C are arrays of characters ending with a null terminator `\0`.

    
#include <stdio.h>
#include <string.h>

int main() {
    // Integer array
    int numbers[5] = {10, 20, 30, 40, 50};
    printf("Third element: %d\n", numbers[2]);
    
    // String (character array)
    char greeting[] = "Hello"; // Null terminator is added automatically
    char name[10] = {'F', 'o', 'r', 'g', 'e', '\0'};

    printf("%s, %s!\n", greeting, name);
    printf("Length of greeting: %zu\n", strlen(greeting));
    
    return 0;
}

    6. Structs

    Structs allow you to group variables of different types under a single name.

    
#include <stdio.h>
#include <string.h>

struct Person {
    char name[50];
    int age;
};

int main() {
    struct Person p1;
    
    strcpy(p1.name, "Alice");
    p1.age = 30;
    
    printf("Name: %s, Age: %d\n", p1.name, p1.age);
    
    // Pointers to structs
    struct Person *p2 = &p1;
    p2->age = 31; // Arrow operator to access members via pointer
    
    printf("Updated age: %d\n", p1.age);
    
    return 0;
}

    7. Memory Management

    C requires manual memory management on the heap using functions from `stdlib.h`.

    
#include <stdio.h>
#include <stdlib.h>

int main() {
    int *arr;
    int n = 5;

    // Allocate memory for 5 integers on the heap
    arr = (int*) malloc(n * sizeof(int));

    if (arr == NULL) {
        printf("Memory allocation failed!\n");
        return 1;
    }

    // Use the allocated memory
    for (int i = 0; i < n; i++) {
        arr[i] = i + 1;
        printf("%d ", arr[i]);
    }
    printf("\n");

    // Free the allocated memory to prevent memory leaks
    free(arr);

    return 0;
}

    9. Preprocessor Directives

    The C preprocessor modifies the source code before compilation. Directives start with `#`.

    
// 1. Include header files for standard library functions
#include <stdio.h> 
#include <stdlib.h>

#define PI 3.14159

    1. Simple Calculator

    A basic calculator that takes an operator and two operands from the user.

    
#include <stdio.h>

int main() {
    char operator;
    double n1, n2;

    printf("Enter an operator (+, -, *, /): ");
    scanf(" %c", &operator);
    printf("Enter two operands: ");
    scanf("%lf %lf", &n1, &n2);

    switch (operator) {
        case '+':
            printf("%.1lf + %.1lf = %.1lf\n", n1, n2, n1 + n2);
            break;
        case '-':
            printf("%.1lf - %.1lf = %.1lf\n", n1, n2, n1 - n2);
            break;
        case '*':
            printf("%.1lf * %.1lf = %.1lf\n", n1, n2, n1 * n2);
            break;
        case '/':
            if (n2 != 0)
                printf("%.1lf / %.1lf = %.1lf\n", n1, n2, n1 / n2);
            else
                printf("Error! Division by zero.\n");
            break;
        default:
            printf("Error! Operator is not correct\n");
    }
    return 0;
}

    Code Explanation

    • scanf: Reads formatted input from the standard input (keyboard). The space before `%c` is important to consume any leftover whitespace.
    • switch: A control flow statement that allows a variable to be tested for equality against a list of values (cases).
    • Error Handling: Includes a basic check to prevent division by zero, a common runtime error.

    2. File Copier

    A command-line program that copies the content of one file to another.

    
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Usage: %s <source_file> <destination_file>\n", argv[0]);
        return 1;
    }

    FILE *source = fopen(argv[1], "rb");
    if (source == NULL) {
        perror("Error opening source file");
        return 1;
    }

    FILE *dest = fopen(argv[2], "wb");
    if (dest == NULL) {
        perror("Error opening destination file");
        fclose(source);
        return 1;
    }

    char buffer[1024];
    size_t bytes_read;
    while ((bytes_read = fread(buffer, 1, sizeof(buffer), source)) > 0) {
        fwrite(buffer, 1, bytes_read, dest);
    }

    printf("File copied successfully.\n");
    fclose(source);
    fclose(dest);
    return 0;
}

    Code Explanation

    • argc & argv: Parameters to `main` that hold the count of command-line arguments and an array of the argument strings, respectively.
    • fopen: Opens a file. `"rb"` and `"wb"` are used for reading and writing in binary mode, which is safe for all file types.
    • fread & fwrite: Read and write blocks of binary data from/to a file stream. The loop continues as long as `fread` successfully reads bytes.

    3. String Reversal Function

    A function that reverses a string in-place using pointers.

    
#include <stdio.h>
#include <string.h>

void reverse_string(char *str) {
    char *end = str + strlen(str) - 1;
    char temp;

    while (str < end) {
        // Swap characters
        temp = *str;
        *str = *end;
        *end = temp;

        // Move pointers towards the middle
        str++;
        end--;
    }
}

int main() {
    char my_string[] = "SyntaxForge";
    printf("Original: %s\n", my_string);
    reverse_string(my_string);
    printf("Reversed: %s\n", my_string);
    return 0;
}

    Code Explanation

    • Pointer Arithmetic: `char *end = str + strlen(str) - 1;` calculates the memory address of the last character of the string.
    • In-place Swap: Two pointers, `str` (start) and `end`, move towards each other. At each step, the characters they point to are swapped using a temporary variable.
    • Dereferencing: The `*` operator is used to get or set the character value at the memory address the pointer is holding.

    4. Simple Linked List

    An implementation of a basic linked list with functions to add a node and print the list.

    
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

void push(Node **head_ref, int new_data) {
    Node *new_node = (Node*) malloc(sizeof(Node));
    new_node->data = new_data;
    new_node->next = *head_ref;
    *head_ref = new_node;
}

void print_list(Node *node) {
    while (node != NULL) {
        printf("%d -> ", node->data);
        node = node->next;
    }
    printf("NULL\n");
}

int main() {
    Node *head = NULL;
    push(&head, 3);
    push(&head, 2);
    push(&head, 1);
    print_list(head);
    return 0;
}

    Code Explanation

    • typedef: Creates a convenient alias (`Node`) for `struct Node`.
    • Self-referential Struct: The `Node` struct contains a pointer to another `Node`, which is the core of a linked list.
    • malloc: Allocates memory for a new node on the heap.
    • Pointer to Pointer (**head_ref): The `push` function needs to modify the `head` pointer in `main`. To do this, we must pass the *address* of the `head` pointer itself.

    5. Caesar Cipher

    A simple encryption program that shifts letters in a message by a fixed amount.

    
#include <stdio.h>
#include <ctype.h>

void caesar_cipher(char *text, int shift) {
    for (int i = 0; text[i] != '\0'; ++i) {
        char ch = text[i];
        if (isalpha(ch)) {
            char base = islower(ch) ? 'a' : 'A';
            ch = (ch - base + shift) % 26 + base;
        }
        text[i] = ch;
    }
}

int main() {
    char message[] = "Hello World";
    int shift = 3;
    printf("Original message: %s\n", message);
    caesar_cipher(message, shift);
    printf("Encrypted message: %s\n", message);
    return 0;
}

    Code Explanation

    • ctype.h: A standard library with useful character-handling functions like `isalpha` (checks if a character is alphabetic) and `islower`.
    • Character Arithmetic: We convert the character to a 0-25 index by subtracting its base (`'a'` or `'A'`).
    • Modulo Operator (%): The modulo operator is key to making the shift "wrap around" from 'z' back to 'a'.
    • In-place Modification: The function modifies the input string directly without needing to return a new one.
